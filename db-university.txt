Selezionare tutti:

1- studenti nati nel 1990
SELECT *
FROM `students`
WHERE YEAR(`date_of_birth`) = 1990;

2- corsi che valgono più di 10 crediti
SELECT * 
FROM `courses`
WHERE `cfu` > 10;

3- studenti che hanno più di 30 anni
SELECT *
FROM `students` 
WHERE DATE_ADD(`date_of_birth`, INTERVAL 30 YEAR) < CURRENT_DATE();

4- corsi del primo semestre del primo anno di un qualsiasi corso di laurea
SELECT * 
FROM `courses`
WHERE `period` = 'I semestre'
AND `year` = 1;

5- appelli d'esame che avvengono nel pomeriggio (dopo le 14) del 20/06/2020
SELECT * 
FROM `exams`
WHERE `date` = '2020/06/20'
AND `hour`
BETWEEN "14:00:00" AND "23:59:59";

6- corsi di laurea magistrale
SELECT * 
FROM `degrees` 
WHERE `level` = 'magistrale';

Conta:

7- numero di dipartimenti universitari
SELECT COUNT(`id`) as `deps_number`
FROM `departments`;

8- numero di insegnanti che non hanno un numero di telefono
SELECT COUNT(`id`)
FROM `teachers`
WHERE `phone` IS NULL;
